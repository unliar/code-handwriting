class LRU {
  constructor(size) {
    this.cache = new Map();
    this.size = size;
  }
  get(k) {
    if (!this.cache.has(k)) return null;
    const v = this.cache.get(k);
    this.cache.delete(k);
    this.cache.set(k, v);
    return v;
  }
  put(k, v) {
    if (this.cache.has(k)) {
      this.cache.delete(k);
    }
    this.cache.set(k, v);
    if (this.cache.size > this.size) {
      const top = this.cache.keys().next().value;
      this.cache.delete(top);
    }
  }
}

  let timer = null;

}

    this.runningCount = 0;
    // æ¶ˆè´¹é˜Ÿåˆ—
    this.queue = [];
    // ç»“æœå€¼
    this.results = [];
  }

  all(promises = []) {
    return new Promise((resolve, reject) => {
      for (const p of promises) {
        this._run(p, resolve, reject);
      }
    });
  }

  _run(promise, resolve, reject) {
    if (this.runningCount >= this.limit) {
      this.queue.push(promise);
      return;
    }

    ++this.runningCount;
    promise
      .then((res) => {
        this.results.push(res);
        --this.runningCount;
        // é˜Ÿåˆ—ä¸º0 ä¸” runnin ä¸º 0 è¯´æ˜ç»“æŸäº†

        if (this.queue.length) {
          this._run(this.queue.shift(), resolve, reject);
        }
      })
      .catch(reject);
  }
}

                                      function PromiseAll(args = []) {
  const fullfilled = [];
  if (args.length == 0) return Promise.resolve(fullfilled);

  return new Promise((resolve, reject) => {
    args.forEach((item, index) => {
      Promise.resolve(item)
        .then((r) => {
          fullfilled[index] = r;
          if (fullfilled.length === args.length) return resolve(fullfilled);
        })
        .catch((e) => reject(e));
    });
  });
}

class LM {
  constructor(name) {
    this.name = name;
    this.q = [];
    this.timer = null;
    this._hi();
  }
  exec() {
    clearTimeout(this.timer);
    this.timer = setTimeout(async () => {
      for (let index = 0; index < this.q.length; index++) {
        await this.q[index]();
      }
    });
    return this;
  }
  sleep(n) {
    this.q.push(() => this._wait(n));
    return this.exec();
  }
  sleepFirst(n) {
    this.q.unshift(() => this._wait(n));
    return this.exec();
  }
  eat(str) {
    this.q.push(() => {
      console.log("eat", str);
    });
    return this.exec();
  }
  _hi() {
    this.q.push(() => {
      console.log("hi", this.name);
    });
    return this.exec();
  }
  _wait(time) {
    return new Promise((y) => {
      setTimeout(() => {
        console.log("wake up after " + time + "s");
        y();
      }, time * 1000);
    });
  }
}

const LazyMan = (name) => {
  return new LM(name);
};

const DeepClone = (src) => {
  let res = null;
  // éæ•°ç»„ æˆ–è€… å¯¹è±¡

    res = src;

      }
    }
  }
  return res;
};
